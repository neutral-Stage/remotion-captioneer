Usage

import { AbsoluteFill } from "remotion";
import { AnimatedCaptions } from "remotion-captioneer";
import captions from "./my-captions.json";

export const MyVideo = () => (
  <AbsoluteFill style={{ background: "#0a0a0a" }}>
    <AnimatedCaptions
      captions={captions}
      style="karaoke"
      highlightColor="#FF6B6B"
    />
  </AbsoluteFill>
);
